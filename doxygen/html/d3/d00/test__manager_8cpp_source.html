190  std::vector<detail::topological_sort_object_t *> &out) {
191  typedef UTIL_UNIT_TEST_MACRO_AUTO_MAP(std::string, detail::topological_sort_object_t) index_by_name_t;
192  out.reserve(in.size());
193 
194  for (index_by_name_t::iterator iter = in.begin(); iter != in.end(); ++iter) {
195  if (0 == iter->second.dependency_count) {
196  out.push_back(&iter->second);
197  }
198  }
199 
200  for (size_t i = 0; i < out.size(); ++i) {
201  for (std::list<detail::topological_sort_object_t *>::iterator iter = out[i]->depend_by.begin();
202  iter != out[i]->depend_by.end(); ++iter) {
203  if ((*iter)->dependency_count > 0) {
204  --(*iter)->dependency_count;
205 
206  if (0 == (*iter)->dependency_count) {
207  out.push_back(*iter);

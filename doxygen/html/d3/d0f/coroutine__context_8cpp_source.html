283 LIBCOPP_COPP_API int coroutine_context::create(coroutine_context *p, callback_t &runner,
284  const stack_context &callee_stack, size_t coroutine_size,
285  size_t private_buffer_size) LIBCOPP_MACRO_NOEXCEPT {
286  if (UTIL_CONFIG_NULLPTR == p) {
287  return COPP_EC_ARGS_ERROR;
288  }
289 
290  // must aligned to sizeof(size_t)
291  if (0 != (private_buffer_size & (sizeof(size_t) - 1))) {
292  return COPP_EC_ARGS_ERROR;
293  }
294 
295  if (0 != (coroutine_size & (sizeof(size_t) - 1))) {
296  return COPP_EC_ARGS_ERROR;
297  }
298 
299  size_t stack_offset = align_stack_size(private_buffer_size + coroutine_size);
300  if (UTIL_CONFIG_NULLPTR == callee_stack.sp || callee_stack.size <= stack_offset) {
301  return COPP_EC_ARGS_ERROR;
302  }
303 
304  // stack down
305  // |STACK BUFFER........COROUTINE..this..padding..PRIVATE DATA.....callee_stack.sp |
306  // |------------------------------callee_stack.size -------------------------------|
307  if (callee_stack.sp <= p || coroutine_size < sizeof(coroutine_context)) {
308  return COPP_EC_ARGS_ERROR;
309  }
310 
311  size_t this_offset = reinterpret_cast<unsigned char *>(callee_stack.sp) - reinterpret_cast<unsigned char *>(p);
312  if (this_offset < sizeof(coroutine_context) + private_buffer_size || this_offset > stack_offset) {
313  return COPP_EC_ARGS_ERROR;
314  }
315 
316  // if runner is empty, we can set it later
317  p->set_runner(std::move(runner));
318 
319  if (&p->callee_stack_ != &callee_stack) {
320  p->callee_stack_ = callee_stack;
321  }
322  p->private_buffer_size_ = private_buffer_size;
323 
324  // stack down, left enough private data
325  p->priv_data_ = reinterpret_cast<unsigned char *>(p->callee_stack_.sp) - p->private_buffer_size_;
326  p->callee_ = fcontext::copp_make_fcontext(reinterpret_cast<unsigned char *>(p->callee_stack_.sp) - stack_offset,
327  p->callee_stack_.size - stack_offset,
328  &libcopp_inner_api_helper::coroutine_context_callback);
329  if (UTIL_CONFIG_NULLPTR == p->callee_) {
330  return COPP_EC_FCONTEXT_MAKE_FAILED;
331  }
332 
333  return COPP_EC_SUCCESS;
334 }

421 LIBCOPP_COPP_API int coroutine_context::yield(void **priv_data) LIBCOPP_MACRO_NOEXCEPT {
422  if (UTIL_CONFIG_NULLPTR == callee_) {
423  return COPP_EC_NOT_INITED;
424  }
425 
426  int from_status = status_t::EN_CRS_RUNNING;
427  int to_status = status_t::EN_CRS_READY;
428  if (check_flags(flag_t::EN_CFT_FINISHED)) {
429  to_status = status_t::EN_CRS_FINISHED;
430  }
431  if (false == status_.compare_exchange_strong(from_status, to_status, libcopp::util::lock::memory_order_acq_rel,
432  libcopp::util::lock::memory_order_acquire)) {
433  switch (from_status) {
434  case status_t::EN_CRS_INVALID:
435  return COPP_EC_NOT_INITED;
436  case status_t::EN_CRS_READY:
437  return COPP_EC_NOT_RUNNING;
438  case status_t::EN_CRS_FINISHED:
439  case status_t::EN_CRS_EXITED:
440  return COPP_EC_ALREADY_EXIST;
441  default:
442  return COPP_EC_UNKNOWN;
443  }
444  }
445 
446  // success or finished will continue
447  jump_src_data_t jump_data;
448  jump_data.from_co = this;
449  jump_data.to_co = UTIL_CONFIG_NULLPTR;
450 
451 #ifdef LIBCOPP_MACRO_USE_SEGMENTED_STACKS
452  jump_to(caller_, callee_stack_, caller_stack_, jump_data);
453 #else
454  jump_to(caller_, callee_stack_, callee_stack_, jump_data);
455 #endif
456 
457  if (UTIL_CONFIG_NULLPTR != priv_data) {
458  *priv_data = jump_data.priv_data;
459  }
460 
461  return COPP_EC_SUCCESS;
462 }

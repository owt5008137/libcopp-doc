73  while (!g_executor.empty()) {
74  example_result_message_t msg;
75  // set a result message

78  // if both success type and error type is a small trivial type, storage_type will be result_t with union of success type and error type
79  // else storage_type will be std::unique_ptr<result_t>
80  // result_t::make_success(...) and result_t::make_error(...) will make sure to use the correct storage type
81  example_result_t::storage_type result_storage = example_result_t::make_success(msg);
82  (*g_executor.begin())->get_private_data().result = &result_storage;
83  // just call context_t::wake to wakeup and poll again
84  (*g_executor.begin())->wake();

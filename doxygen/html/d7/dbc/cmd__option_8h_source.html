221  static const char *get_segment(const char *begin_str, std::string &val) {
222  val.clear();
223  char flag; // 字符串开闭字符
224 
225  // 去除分隔符前缀
226  while (*begin_str && (map_value_[(uc_t)*begin_str] & SPLITCHAR)) ++begin_str;
227 
228  while (*begin_str && !(map_value_[(uc_t)*begin_str] & SPLITCHAR)) {
229  if (!(map_value_[(uc_t)*begin_str] & STRINGSYM)) {

255  while (*begin_str && (map_value_[(uc_t)*begin_str] & SPLITCHAR)) ++begin_str;

377  void start(callback_param args, bool is_single_cmd = false) const {
378  int argv = static_cast<int>(args.get_params_number());
379  cmd_option_list cmd_args;
380  TCmdStr cmd_content = is_single_cmd ? "@OnError" : "@OnDefault";
381  for (int i = -1; i < argv;) {
382  ++i;
383  cmd_args.clear();
384  cmd_args.load_cmd_array(args.get_cmd_array());
385  cmd_args.set_ext_param(args.get_ext_param());
386 
387  for (; i < argv; ++i) {
388  // 把所有的非指令字符串设为指令参数
389  if (callback_funcs_.find(args[i]->to_string()) == callback_funcs_.end()) {
390  cmd_args.add(args[i]->to_string());
391  } else {
392  // 如果是单指令且有未知参数则分发@OnError错误处理
393  if (is_single_cmd && cmd_args.get_params_number() > 0) {
394  run_cmd(cmd_content, cmd_args);
395  cmd_args.clear();
396  cmd_args.load_cmd_array(args.get_cmd_array());
397  cmd_args.set_ext_param(args.get_ext_param());
398  }
399 
400  // 追加所有参数，执行单指令
401  if (is_single_cmd) {
402  cmd_content = TCmdStr(args[i]->to_cpp_string().c_str(), args[i]->to_cpp_string().size());
403  for (++i; i < argv; ++i) cmd_args.add(args[i]->to_string());
404  }
405  break;
406  }
407  }

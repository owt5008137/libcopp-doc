314  if (false == status_.compare_exchange_strong(from_status, to_status, libcopp::util::lock::memory_order_acq_rel,

317  case status_t::EN_CRS_INVALID:
318  return COPP_EC_NOT_INITED;
319  case status_t::EN_CRS_READY:
320  return COPP_EC_NOT_RUNNING;
321  case status_t::EN_CRS_FINISHED:
322  case status_t::EN_CRS_EXITED:
323  return COPP_EC_ALREADY_EXIST;

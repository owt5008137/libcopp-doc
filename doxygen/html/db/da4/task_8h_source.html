405  inline int await_task(ptr_t wait_task) {
406  if (!wait_task) {
407  return copp::COPP_EC_ARGS_ERROR;
408  }
409 
410  if (this == wait_task.get()) {
411  return copp::COPP_EC_TASK_CAN_NOT_WAIT_SELF;
412  }
413 
414  // if target is exiting or completed, just return
415  if (wait_task->is_exiting() || wait_task->is_completed()) {
416  return copp::COPP_EC_TASK_IS_EXITING;
417  }
418 
419  if (is_exiting()) {
420  return copp::COPP_EC_TASK_IS_EXITING;
421  }
422 
423  if (this_task() != this) {
424  return copp::COPP_EC_TASK_NOT_IN_ACTION;
425  }
426 
427  // add to next list failed
428  if (wait_task->next(ptr_t(this)).get() != this) {
429  return copp::COPP_EC_TASK_ADD_NEXT_FAILED;
430  }
431 
432  int ret = 0;
433  while (!(wait_task->is_exiting() || wait_task->is_completed())) {
434  if (is_exiting()) {
435  return copp::COPP_EC_TASK_IS_EXITING;
436  }
437 
438  ret = yield();
439  }
440 
441  return ret;
442  }

535  virtual int start(std::list<std::exception_ptr> &unhandled, void *priv_data,
536  EN_TASK_STATUS expected_status = EN_TS_CREATED) LIBCOPP_MACRO_NOEXCEPT {
537 #else
538  virtual int start(void *priv_data, EN_TASK_STATUS expected_status = EN_TS_CREATED) UTIL_CONFIG_OVERRIDE {
539 #endif
540  if (!coroutine_obj_) {
541  return copp::COPP_EC_NOT_INITED;
542  }
543 
544  EN_TASK_STATUS from_status = expected_status;
545 
546  do {
547  if (unlikely(from_status >= EN_TS_DONE)) {
548  return copp::COPP_EC_ALREADY_FINISHED;
549  }
550 
551  if (unlikely(from_status == EN_TS_RUNNING)) {
552  return copp::COPP_EC_IS_RUNNING;
553  }
554 
555  if (likely(_cas_status(from_status, EN_TS_RUNNING))) { // Atomic.CAS here
556  break;
557  }
558  } while (true);
559 
560  // use this smart ptr to avoid destroy of this
561  // ptr_t protect_from_destroy(this);
562 
563 #if defined(LIBCOPP_MACRO_ENABLE_STD_EXCEPTION_PTR) && LIBCOPP_MACRO_ENABLE_STD_EXCEPTION_PTR
564  std::exception_ptr eptr;
565  int ret = coroutine_obj_->start(eptr, priv_data);
566  if (eptr) {
567  unhandled.emplace_back(std::move(eptr));
568  }
569 #else
570  int ret = coroutine_obj_->start(priv_data);
571 #endif
572 
573  from_status = EN_TS_RUNNING;
574  if (is_completed()) { // Atomic.CAS here
575  while (from_status < EN_TS_DONE) {
576  if (likely(_cas_status(from_status, EN_TS_DONE))) { // Atomic.CAS here
577  break;
578  }
579  }
580 
581  finish_priv_data_ = priv_data;
582 #if defined(LIBCOPP_MACRO_ENABLE_STD_EXCEPTION_PTR) && LIBCOPP_MACRO_ENABLE_STD_EXCEPTION_PTR
583  _notify_finished(unhandled, priv_data);
584 #else
585  _notify_finished(priv_data);
586 #endif
587  return ret;
588  }
589 
590  while (true) {
591  if (from_status >= EN_TS_DONE) { // canceled or killed
592 #if defined(LIBCOPP_MACRO_ENABLE_STD_EXCEPTION_PTR) && LIBCOPP_MACRO_ENABLE_STD_EXCEPTION_PTR
593  _notify_finished(unhandled, finish_priv_data_);
594 #else
595  _notify_finished(finish_priv_data_);
596 #endif
597  break;
598  }
599 
600  if (likely(_cas_status(from_status, EN_TS_WAITING))) { // Atomic.CAS here
601  break;
602  // waiting
603  }
604  }
605 
606  return ret;
607  }

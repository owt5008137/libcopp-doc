291  inline ptr_t next(ptr_t next_task, void *priv_data = UTIL_CONFIG_NULLPTR) {
292  // can not refers to self
293  if (this == next_task.get() || !next_task) {
294  return ptr_t(this);
295  }
296 
297  // can not add next task when finished
298  if (is_exiting() || is_completed()) {
299  // run next task immedialy
300  EN_TASK_STATUS next_task_status = next_task->get_status();
301  if (EN_TS_CREATED == next_task_status) {
302  next_task->start(priv_data);
303  } else if (EN_TS_WAITING == next_task_status) {
304  next_task->resume(priv_data);
305  }
306  return next_task;
307  }

398  inline int await_task(ptr_t wait_task) {
399  if (!wait_task) {
400  return copp::COPP_EC_ARGS_ERROR;
401  }
402 
403  if (this == wait_task.get()) {
404  return copp::COPP_EC_TASK_CAN_NOT_WAIT_SELF;
405  }
406 
407  // if target is exiting or completed, just return
408  if (wait_task->is_exiting() || wait_task->is_completed()) {
409  return copp::COPP_EC_TASK_IS_EXITING;
410  }
411 
412  if (is_exiting()) {
413  return copp::COPP_EC_TASK_IS_EXITING;
414  }
415 
416  if (this_task() != this) {
417  return copp::COPP_EC_TASK_NOT_IN_ACTION;
418  }
419 
420  // add to next list failed
421  if (wait_task->next(ptr_t(this)).get() != this) {
422  return copp::COPP_EC_TASK_ADD_NEXT_FAILED;
423  }
424 
425  int ret = 0;
426  while (!(wait_task->is_exiting() || wait_task->is_completed())) {
427  if (is_exiting()) {
428  return copp::COPP_EC_TASK_IS_EXITING;
429  }
430 
431  ret = yield();
432  }
433 
434  return ret;
435  }

526  virtual int start(std::list<std::exception_ptr> &unhandled, void *priv_data,
527  EN_TASK_STATUS expected_status = EN_TS_CREATED) LIBCOPP_MACRO_NOEXCEPT {
528 #else
529  virtual int start(void *priv_data, EN_TASK_STATUS expected_status = EN_TS_CREATED) UTIL_CONFIG_OVERRIDE {
530 #endif
531  if (!coroutine_obj_) {
532  return copp::COPP_EC_NOT_INITED;
533  }
534 
535  EN_TASK_STATUS from_status = expected_status;
536 
537  do {
538  if (unlikely(from_status >= EN_TS_DONE)) {
539  return copp::COPP_EC_ALREADY_FINISHED;
540  }
541 
542  if (unlikely(from_status == EN_TS_RUNNING)) {
543  return copp::COPP_EC_IS_RUNNING;
544  }
545 
546  if (likely(_cas_status(from_status, EN_TS_RUNNING))) { // Atomic.CAS here
547  break;
548  }
549  } while (true);
550 
551  // use this smart ptr to avoid destroy of this
552  // ptr_t protect_from_destroy(this);
553 
554 #if defined(LIBCOPP_MACRO_ENABLE_STD_EXCEPTION_PTR) && LIBCOPP_MACRO_ENABLE_STD_EXCEPTION_PTR
555  std::exception_ptr eptr;
556  int ret = coroutine_obj_->start(eptr, priv_data);
557  if (eptr) {
558  unhandled.emplace_back(std::move(eptr));
559  }
560 #else
561  int ret = coroutine_obj_->start(priv_data);
562 #endif
563 
564  from_status = EN_TS_RUNNING;
565  if (is_completed()) { // Atomic.CAS here
566  while (from_status < EN_TS_DONE) {
567  if (likely(_cas_status(from_status, EN_TS_DONE))) { // Atomic.CAS here
568  break;
569  }
570  }
571 
572  finish_priv_data_ = priv_data;
573 #if defined(LIBCOPP_MACRO_ENABLE_STD_EXCEPTION_PTR) && LIBCOPP_MACRO_ENABLE_STD_EXCEPTION_PTR
574  _notify_finished(unhandled, priv_data);
575 #else
576  _notify_finished(priv_data);
577 #endif
578  return ret;
579  }
580 
581  while (true) {
582  if (from_status >= EN_TS_DONE) { // canceled or killed
583 #if defined(LIBCOPP_MACRO_ENABLE_STD_EXCEPTION_PTR) && LIBCOPP_MACRO_ENABLE_STD_EXCEPTION_PTR
584  _notify_finished(unhandled, finish_priv_data_);
585 #else
586  _notify_finished(finish_priv_data_);
587 #endif
588  break;
589  }
590 
591  if (likely(_cas_status(from_status, EN_TS_WAITING))) { // Atomic.CAS here
592  break;
593  // waiting
594  }
595  }
596 
597  return ret;
598  }

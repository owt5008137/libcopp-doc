665  if (flags_ & flag_t::EN_TM_IN_RESET) {
666  return copp::COPP_EC_IN_RESET;
667  }
668 
669  // first tick, init and reset task timeout
670  if (0 == last_tick_time_.tv_sec && 0 == last_tick_time_.tv_nsec) {
671  // hold lock
672 #if !defined(LIBCOPP_DISABLE_ATOMIC_LOCK) || !(LIBCOPP_DISABLE_ATOMIC_LOCK)
673  libcopp::util::lock::lock_holder<libcopp::util::lock::spin_lock> lock_guard(action_lock_);
674 #endif

677  for (typename std::set<detail::task_timer_node<task_t> >::iterator iter = task_timeout_timer_.begin();
678  task_timeout_timer_.end() != iter; ++iter) {
679  detail::task_timer_node<task_t> new_checkpoint = (*iter);
680  new_checkpoint.expired_time.tv_sec += sec;
681  new_checkpoint.expired_time.tv_nsec += nsec;
682  real_checkpoints.insert(new_checkpoint);
683  }
684 
685  task_timeout_timer_.swap(task_timeout_timer_);
686  last_tick_time_ = now_tick_time;
687  return copp::COPP_EC_SUCCESS;
688  }
689 
690 #if defined(LIBCOPP_MACRO_ENABLE_STD_EXCEPTION_PTR) && LIBCOPP_MACRO_ENABLE_STD_EXCEPTION_PTR
691  std::list<std::exception_ptr> eptrs;
692 #endif
693  // remove timeout tasks
694  while (false == task_timeout_timer_.empty()) {
695  task_ptr_t task_inst;
696 
697  {
698  // hold lock
699 #if !defined(LIBCOPP_DISABLE_ATOMIC_LOCK) || !(LIBCOPP_DISABLE_ATOMIC_LOCK)
700  libcopp::util::lock::lock_holder<libcopp::util::lock::spin_lock> lock_guard(action_lock_);
701 #endif
702 
703  const typename std::set<detail::task_timer_node<task_t> >::value_type &timer_node =
704  *task_timeout_timer_.begin();
705  // all tasks those expired time less than now are timeout
706  if (now_tick_time <= timer_node.expired_time) {
707  break;
708  }
709 
710  // check expire time(may be changed)
711  typedef typename container_t::iterator iter_type;
712 
713  iter_type iter = tasks_.find(timer_node.task_id);
714 
715  if (tasks_.end() != iter) {
716  // task may be removed before
717  task_inst = std::move(iter->second.task_);
718 
719  remove_timeout_timer(iter->second);
720  tasks_.erase(iter); // remove from container
721  }
722  }
723 
724  // task call can not be used when lock is on
725  if (task_inst && !task_inst->is_exiting()) {
726 #if defined(LIBCOTASK_MACRO_AUTO_CLEANUP_MANAGER) && LIBCOTASK_MACRO_AUTO_CLEANUP_MANAGER
727  // already cleanup, there is no need to cleanup again
728  task_t::task_manager_helper::cleanup_task_manager(*task_inst, reinterpret_cast<void *>(this));
729 #endif
730 #if defined(LIBCOPP_MACRO_ENABLE_STD_EXCEPTION_PTR) && LIBCOPP_MACRO_ENABLE_STD_EXCEPTION_PTR
731  task_inst->kill(eptrs, EN_TS_TIMEOUT, nullptr);
732 #else
733  task_inst->kill(EN_TS_TIMEOUT);
734 #endif
735  }
736  }
737 
738  last_tick_time_ = now_tick_time;
739 
740 #if defined(LIBCOPP_MACRO_ENABLE_STD_EXCEPTION_PTR) && LIBCOPP_MACRO_ENABLE_STD_EXCEPTION_PTR
741  task_t::maybe_rethrow(eptrs);
742 #endif
743  return copp::COPP_EC_SUCCESS;
744  }

780  remove_timeout_timer(node);
781

799  }

805  node.timer_node = task_timeout_timer_.end();
806  }
807  }

78 static void benchmark_round(int index) {

95  printf("create %d future and context, cost time: %d s, clock time: %d ms, avg: %lld ns\n", max_task_number,
96  static_cast<int>(end_time - begin_time), CALC_MS_CLOCK(end_clock - begin_clock),
97  CALC_NS_AVG_CLOCK(end_clock - begin_clock, max_task_number));
98 
99  begin_time = end_time;
100  begin_clock = end_clock;
101 
102  // yield & resume from runner
103  bool continue_flag = true;
104  long long real_switch_times = static_cast<long long>(0);
105 
106  while (continue_flag) {
107  continue_flag = false;
108  for (int i = 0; i < max_task_number; ++i) {
109  if (false == task_arr[i]->is_ready()) {
110  continue_flag = true;
111  ++real_switch_times;
112  task_arr[i]->poll(*context_arr[i]);
113  }
114  }
115  }
116 
117  end_time = time(NULL);
118  end_clock = CALC_CLOCK_NOW();
119  printf("poll %d future and context for %lld times, cost time: %d s, clock time: %d ms, avg: %lld ns\n",
120  max_task_number, real_switch_times, static_cast<int>(end_time - begin_time),
121  CALC_MS_CLOCK(end_clock - begin_clock), CALC_NS_AVG_CLOCK(end_clock - begin_clock, real_switch_times));
122 
123  begin_time = end_time;
124  begin_clock = end_clock;
125 
126  task_arr.clear();
127  context_arr.clear();
128 
129  end_time = time(NULL);
130  end_clock = CALC_CLOCK_NOW();
131  printf("remove %d future and context, cost time: %d s, clock time: %d ms, avg: %lld ns\n", max_task_number,

14 // TIMESTAMP:32|SEQUENCE:24 -> about 16M id per second
15 static uint64_t allocate_id_by_atomic() {
16  static libcopp::util::lock::atomic_int_type<uint64_t> seq_alloc(0);
17 
18  //
19 #if defined(UTIL_CONFIG_COMPILER_CXX_CONSTEXPR) && UTIL_CONFIG_COMPILER_CXX_CONSTEXPR
20  static constexpr size_t seq_bits = 24;
21  static constexpr uint64_t time_mask = (static_cast<uint64_t>(1) << 32) - 1;
22 #else
23  static const size_t seq_bits = 24;
24  static const uint64_t time_mask = (static_cast<uint64_t>(1) << 32) - 1;
25 #endif
26 
27  // always do not allocate 0 as a valid ID
28  uint64_t ret = 0;
29  while (0 == ret) {
30  uint64_t res = seq_alloc.load();
31  uint64_t time_part = res >> seq_bits;
32 
33  uint64_t next_ret = res + 1;
34  uint64_t next_time_part = next_ret >> seq_bits;
35  if (0 == time_part || time_part != next_time_part) {
36  uint64_t now_time = time_part;
37  while (time_part == now_time) {
38  now_time = (static_cast<uint64_t>(time(NULL)) & time_mask) - 1577836800; // 2020-01-01 00:00:00+00:00 UTC
39  }
40 
41  // if failed, maybe another thread do it
42  if (seq_alloc.compare_exchange_strong(res, now_time << seq_bits, libcopp::util::lock::memory_order_acq_rel,
43  libcopp::util::lock::memory_order_acquire)) {
44  ret = now_time << seq_bits;
45  }
46  } else {
47  if (seq_alloc.compare_exchange_weak(res, next_ret, libcopp::util::lock::memory_order_acq_rel,
48  libcopp::util::lock::memory_order_acquire)) {
49  ret = next_ret;
50  }
51  }
52  }
53 
54  return ret;
55 }

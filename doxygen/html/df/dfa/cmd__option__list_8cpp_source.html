33  typedef std::vector<value_type> keys_type;

39  for (keys_type::size_type i = 0; i < keys_.size(); ++i) {

41  if (NULL == str_key) {
42  continue;
43  }
44

52  }

54  std::string val;
55  cmd_option::get_segment(str_val, val);
